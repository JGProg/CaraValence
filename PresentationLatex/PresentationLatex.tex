\subsection{Son utilité}
\begin{frame}
  \frametitle{L'automatisation}
\framesubtitle{Son utilité}
  \begin{block}{Les apports}
		\begin{itemize}
		\item Permettre une fréquence de test très élevée (intégration continue),
	  \item Garantir un haut niveau de répétabilité,
	  \item Apporter un "formalisme" dans la description des comportements attendus.
	\end{itemize}
	\end{block}
